#include <Keypad.h>
#include <LiquidCrystal.h>

LiquidCrystal miLCD(8,13,A0,A1,A2,A3);

const byte FILAS = 4;
const byte COLUMNAS = 4;



char caracteres [4] [4] =
{
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'},
};

byte pinesFilas[] = {7,6,5,4};
byte pinesColumnas[] = {3,2,1,0};



Keypad keypad = Keypad(makeKeymap(caracteres),pinesFilas,pinesColumnas,FILAS,COLUMNAS);




void setup()
{
  miLCD.begin(16,2);
  pinMode(LED_BUILTIN, OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  miLCD.setCursor(0,0);
  miLCD.print("Password:");
  
  char boton = keypad.getKey();// lee lo pulsado en el teclado o un define de no click
  
  if(boton != NO_KEY)
  {
    miLCD.print(boton);
   
  }

     
  delay(25);
  
  
}
